class RetrievalService:
    @classmethod
    def retrieve(cls, retrieval_method, dataset_id, query, top_k=4, ...):
        # 并行执行多种检索策略
        with ThreadPoolExecutor(max_workers=dify_config.RETRIEVAL_SERVICE_EXECUTORS) as executor:
            futures = []
            
            if query:
                # 向量检索
                futures.append(executor.submit(
                    retrieval_service._retrieve,  # 内部调用 embedding_search
                    ...
                ))
            
            if attachment_ids:
                # 附件检索
                for attachment_id in attachment_ids:
                    futures.append(executor.submit(...))
            
            # 等待所有任务完成
            for _ in concurrent.futures.as_completed(futures, timeout=3600):
                ...
        
        return all_documents
    
    @classmethod
    def embedding_search(cls, flask_app, dataset_id, query, ...):
        # 向量检索核心实现
        vector = Vector(dataset=dataset)
        
        # 混合搜索时特殊处理阈值
        embedding_score_threshold = (
            0.0 if retrieval_method == RetrievalMethod.HYBRID_SEARCH else score_threshold
        )
        
        # 执行向量搜索
        documents = vector.search_by_vector(
            query,
            search_type="similarity_score_threshold",
            top_k=top_k,
            score_threshold=embedding_score_threshold,
            filter={"group_id": [dataset.id]},
            ...
        )
        
        # 如果配置了重排序模型
        if reranking_model and retrieval_method == RetrievalMethod.SEMANTIC_SEARCH:
            data_post_processor = DataPostProcessor(
                str(dataset.tenant_id), 
                str(RerankMode.RERANKING_MODEL), 
                reranking_model, 
                None, 
                False
            )
            documents = data_post_processor.invoke(
                query=query,
                documents=documents,
                score_threshold=score_threshold,
                top_n=len(documents),
            )
        
        return documents
    
    @classmethod
    def keyword_search(cls, flask_app, dataset_id, query, ...):
        # 关键词检索实现
        keyword = Keyword(dataset=dataset)
        
        # 转义查询字符串
        escaped_query = cls.escape_query_for_search(query)
        
        documents = keyword.search(
            escaped_query,
            top_k=top_k,
            document_ids_filter=document_ids_filter
        )
        
        return documents
    
    @classmethod
    def full_text_index_search(cls, flask_app, dataset_id, query, ...):
        # 全文索引检索 (ES)
        vector_processor = Vector(dataset=dataset)
        
        documents = vector_processor.search_by_full_text(
            cls.escape_query_for_search(query),
            top_k=top_k,
            document_ids_filter=document_ids_filter
        )
        
        return documents

            self._param.top_n,                  # top_n

            doc_ids=doc_ids,
            aggs=True,
            rerank_mdl=rerank_mdl,
            rank_feature=label_question(query, kbs),
        )
        
        # 7. TOC 增强检索 (可选)
        if self._param.toc_enhance:
            cks = await settings.retriever.retrieval_by_toc(
                query, kbinfos["chunks"], [kb.tenant_id for kb in kbs],
                chat_mdl, self._param.top_n
            )

                kbinfos["chunks"] = cks

            kbinfos["chunks"], [kb.tenant_id for kb in kbs]

            ck = await settings.kg_retriever.retrieval(
                query, [kb.tenant_id for kb in kbs], kb_ids,
                embd_mdl, LLMBundle(tenant_id, chat_model_config)
            )

                kbinfos["chunks"].insert(0, ck)

        for ck in kbinfos["chunks"]:
            if "vector" in ck:
                del ck["vector"]
            if "content_ltks" in ck:
                del ck["content_ltks"]

### 3.1 父子检索器
**源码位置**: `QAnything/qanything_kernel/core/retriever/parent_retriever.py`

```python
class ParentRetriever:
    async def get_retrieved_documents(
        self, 
        query: str, 
        partition_keys: List[str], 
        time_record: dict,
        hybrid_search: bool, 
        top_k: int
    ):
        # 1. Milvus 向量检索
        expr = f'kb_id in {partition_keys}'
        self.retriever.set_search_kwargs("similarity", k=top_k, expr=expr)
        
        # 2. 执行检索获取子文档
        query_docs = await self.retriever.aget_relevant_documents(query)
        
        # 3. 标记来源
        for doc in query_docs:
            doc.metadata['retrieval_source'] = 'milvus'
        
        # 4. 如果需要混合搜索
        if not hybrid_search:
            return query_docs
        
        # 5. Elasticsearch 关键词检索
        try:
            filter = [{"terms": {"metadata.kb_id.keyword": partition_keys}}]
            es_sub_docs = await self.es_store.asimilarity_search(
                query, k=top_k, filter=filter
            )
            
            # 6. 获取 Milvus 中没有的文档
            milvus_doc_ids = [d.metadata[self.retriever.id_key] for d in query_docs]
            es_ids = []
            for d in es_sub_docs:
                if self.retriever.id_key in d.metadata:
                    if d.metadata[self.retriever.id_key] not in milvus_doc_ids:
                        es_ids.append(d.metadata[self.retriever.id_key])
            
            # 7. 从文档存储获取 ES 匹配的父文档
            es_docs = await self.retriever.docstore.amget(es_ids)
            es_docs = [d for d in es_docs if d is not None]
            
            # 8. 标记来源并合并
            for doc in es_docs:
                doc.metadata['retrieval_source'] = 'es'
            
            query_docs.extend(es_docs)
        except Exception as e:
            debug_logger.error(f"Error in get_retrieved_documents on es_search: {e}")
        
        return query_docs
```

@classmethod
def retrieve(cls, retrieval_method, dataset_id, query, top_k=4, ...):
    # 使用线程池并行执行
    with ThreadPoolExecutor(max_workers=dify_config.RETRIEVAL_SERVICE_EXECUTORS) as executor:
        futures = []
        
        if query:

                retrieval_method=retrieval_method,

        # 等待所有任务完成

            if exceptions:
                for f in futures:
                    f.cancel()
                break
        
        return all_documents

## 6. 性能优化对比

### 6.1 Dify 性能优化
- **ThreadPoolExecutor 并行**: 多线程并行执行多种检索
- **批量处理**: 向量批量 embedding (batch_size=1000)
- **正则缓存**: 预编译正则表达式避免重复编译

### 6.2 RAGFlow 性能优化
- **加权融合**: 0.05 关键词 + 0.95 向量的高效融合
- **降级重试**: 无结果时自动降低阈值重试
- **异步执行**: async/await 异步检索

### 6.3 QAnything 性能优化
- **父子分离**: 只索引子文档，减少向量维度
- **分区检索**: 使用 Milvus 分区提高检索效率
- **结果缓存**: 缓存检索结果
