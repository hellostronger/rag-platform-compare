#### 1.2.1 段落索引处理器
**源码位置**: `dify/api/core/rag/index_processor/processor/paragraph_index_processor.py`

```python
class ParagraphIndexProcessor(BaseIndexProcessor):
    def extract(self, extract_setting: ExtractSetting, **kwargs) -> list[Document]:
        """文档提取"""
        text_docs = ExtractProcessor.extract(
            extract_setting=extract_setting,
            is_automatic=(
                kwargs.get("process_rule_mode") == "automatic" 
                or kwargs.get("process_rule_mode") == "hierarchical"
            ),
        )
        return text_docs
    
    def transform(self, documents: list[Document], current_user: Account | None = None, **kwargs) -> list[Document]:
        """文本分块"""
        process_rule = kwargs.get("process_rule")
        
        if process_rule.get("mode") == "automatic":
            automatic_rule = DatasetProcessRule.AUTOMATIC_RULES
            rules = Rule.model_validate(automatic_rule)
        else:
            rules = Rule.model_validate(process_rule.get("rules"))
        
        # 创建分块器
        splitter = self._get_splitter(
            processing_rule_mode=process_rule.get("mode"),
            max_tokens=rules.segmentation.max_tokens,
            chunk_overlap=rules.segmentation.chunk_overlap,
            separator=rules.segmentation.separator,
            embedding_model_instance=kwargs.get("embedding_model_instance"),
        )
        
        all_documents = []
        for document in documents:
            # 文档清洗
            document_text = CleanProcessor.clean(document.page_content, kwargs.get("process_rule", {}))
            document.page_content = document_text
            
            # 分块
            document_nodes = splitter.split_documents([document])
            
            for document_node in document_nodes:
                if document_node.page_content.strip():
                    doc_id = str(uuid.uuid4())
                    hash = helper.generate_text_hash(document_node.page_content)
                    document_node.metadata["doc_id"] = doc_id
                    document_node.metadata["doc_hash"] = hash
                    
                    # 获取多模态内容
                    multimodal_documents = self._get_content_files(document_node, current_user)
                    if multimodal_documents:
                        document_node.attachments = multimodal_documents
                    
                    # 清理分块字符
                    page_content = remove_leading_symbols(document_node.page_content).strip()
                    if len(page_content) > 0:
                        document_node.page_content = page_content
                        split_documents.append(document_node)
            
            all_documents.extend(split_documents)
        
        return all_documents
    
    def load(self, dataset: Dataset, documents: list[Document], multimodal_documents=None, with_keywords=True, **kwargs):
        """创建索引"""
        # 向量索引 (高质量模式)
        if dataset.indexing_technique == IndexTechniqueType.HIGH_QUALITY:
            vector = Vector(dataset)
            vector.create(documents)
            
            # 多模态索引
            if multimodal_documents and dataset.is_multimodal:
                vector.create_multimodal(multimodal_documents)
        
        # 关键词索引
        if with_keywords:
            keywords_list = kwargs.get("keywords_list")
            keyword = Keyword(dataset)
            if keywords_list and len(keywords_list) > 0:
                keyword.add_texts(documents, keywords_list=keywords_list)
            else:
                keyword.add_texts(documents)
```

#### 1.2.2 父子索引处理器
**源码位置**: `dify/api/core/rag/index_processor/processor/parent_child_index_processor.py`

```python
class ParentChildIndexProcessor(BaseIndexProcessor):
    def transform(self, documents: list[Document], current_user: Account | None = None, **kwargs) -> list[Document]:
        rules = Rule.model_validate(process_rule.get("rules"))
        
        all_documents = []
        
        if rules.parent_mode == ParentMode.PARAGRAPH:
            # 段落模式: 先分段再创建子节点
            splitter = self._get_splitter(...)
            
            for document in documents:
                # 文档清洗
                document_text = CleanProcessor.clean(document.page_content, process_rule)
                document.page_content = document_text
                
                # 父块分割
                document_nodes = splitter.split_documents([document])
                
                for document_node in document_nodes:
                    # 创建子节点
                    child_nodes = self._split_child_nodes(
                        document_node, rules, process_rule.get("mode"), 
                        kwargs.get("embedding_model_instance")
                    )
                    document_node.children = child_nodes
                    split_documents.append(document_node)
                
                all_documents.extend(split_documents)
        
        elif rules.parent_mode == ParentMode.FULL_DOC:
            # 全文模式: 整个文档作为父块
            page_content = "\n".join([document.page_content for document in documents])
            document = Document(page_content=page_content, metadata=documents[0].metadata)
            
            # 创建子节点
            child_nodes = self._split_child_nodes(document, rules, ...)
            document.children = child_nodes
            
            doc_id = str(uuid.uuid4())
            hash = helper.generate_text_hash(document.page_content)
            document.metadata["doc_id"] = doc_id
            document.metadata["doc_hash"] = hash
            
            all_documents.append(document)
        
        return all_documents
    
    def load(self, dataset: Dataset, documents: list[Document], ...):
        """只索引子文档"""
        if dataset.indexing_technique == IndexTechniqueType.HIGH_QUALITY:
            vector = Vector(dataset)
            for document in documents:
                child_documents = document.children
                if child_documents:
                    # 格式化子文档
                    formatted_child_documents = [
                        Document.model_validate(child_document.model_dump()) 
                        for child_document in child_documents
                    ]
                    # 只为子文档创建向量索引
                    vector.create(formatted_child_documents)
    
    def _split_child_nodes(self, document, rules, processing_rule_mode, embedding_model_instance):
        """分割子节点"""
        # 子块大小通常是父块的一半
        child_max_tokens = rules.segmentation.max_tokens // 2
        child_chunk_overlap = rules.segmentation.chunk_overlap
        
        child_splitter = self._get_splitter(
            processing_rule_mode=processing_rule_mode,
            max_tokens=child_max_tokens,
            chunk_overlap=child_chunk_overlap,
            separator=rules.segmentation.separator,
            embedding_model_instance=embedding_model_instance,
        )
        
        child_nodes = child_splitter.split_documents([document])
        
        # 设置子节点元数据
        for i, child_node in enumerate(child_nodes):
            child_node.metadata["child_index"] = i
        
        return child_nodes
```

    # 1. 获取数据集和文档

# QAnything/qanything_kernel/core/retriever/parent_retriever.py:69-156
async def aadd_documents(self, documents, ids=None, add_to_docstore=True, ...):
    # 1. 父块分割
    if self.parent_splitter is not None and not single_parent:
        split_documents = []
        for doc in documents:
            if doc.metadata['has_table'] or num_tokens_embed(doc.page_content) <= parent_chunk_size:
                split_documents.append(doc)
            else:
                need_split_docs.append(doc)
        if need_split_docs:
            split_documents.extend(self.parent_splitter.split_documents(need_split_docs))
        documents = split_documents
    
    # 2. 子块分割
    docs = []
    full_docs = []
    for doc in documents:
        _id = doc_ids[i]
        sub_docs = self.child_splitter.split_documents([doc])
        
        # 设置子块元数据，关联父文档 ID
        for _doc in sub_docs:
            _doc.metadata[self.id_key] = _id
            _doc.page_content = f"[headers]({_doc.metadata['headers']})\n" + _doc.page_content
        
        docs.extend(sub_docs)
        full_docs.append((_id, doc))
    
    # 3. 创建向量索引 (子块)
    embed_docs = copy.deepcopy(docs)
    # 清理不需要的 metadata
    for doc in embed_docs:
        del doc.metadata['title_lst']
        del doc.metadata['has_table']
        del doc.metadata['images']
        ...
    
    res = await self.vectorstore.aadd_documents(embed_docs)
    
    # 4. 创建 ES 全文索引
    if es_store is not None:
        docs_ids = [doc.metadata['file_id'] + '_' + str(i) for i, doc in enumerate(embed_docs)]
        es_res = await es_store.aadd_documents(embed_docs, ids=docs_ids)
    
    # 5. 存储父文档到 MySQL
    if add_to_docstore:
        await self.docstore.amset(full_docs)
    
    return len(res), time_record
