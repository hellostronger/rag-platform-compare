```python
class RetrievalService:
    @classmethod
    def retrieve(cls, retrieval_method, dataset_id, query, top_k=4, ...):
        # 1. 获取数据集
        dataset = cls._get_dataset(dataset_id)
        
        # 2. 使用线程池并行执行
        with ThreadPoolExecutor(max_workers=dify_config.RETRIEVAL_SERVICE_EXECUTORS) as executor:
            futures = []
            
            # 向量检索任务
            if query:
                futures.append(executor.submit(
                    cls.embedding_search,
                    flask_app=current_app._get_current_object(),
                    retrieval_method=retrieval_method,
                    dataset=dataset,
                    query=query,
                    ...
                ))
            
            # 附件检索任务
            if attachment_ids:
                for attachment_id in attachment_ids:
                    futures.append(executor.submit(...))
        
        # 3. 等待所有任务完成
        for _ in concurrent.futures.as_completed(futures, timeout=3600):
            ...
        
        # 4. 去重
        all_documents = cls._deduplicate_documents(all_documents)
        
        return all_documents
    
    @classmethod
    def embedding_search(cls, flask_app, dataset_id, query, ...):
        with flask_app.app_context():
            dataset = cls._get_dataset(dataset_id)
            vector = Vector(dataset=dataset)
            
            # 向量搜索
            documents = vector.search_by_vector(
                query,
                search_type="similarity_score_threshold",
                top_k=top_k,
                score_threshold=embedding_score_threshold,
                filter={"group_id": [dataset.id]},
                document_ids_filter=document_ids_filter,
            )
            
            # 重排序
            if reranking_model and retrieval_method == RetrievalMethod.SEMANTIC_SEARCH:
                data_post_processor = DataPostProcessor(...)
                documents = data_post_processor.invoke(
                    query=query,
                    documents=documents,
                    score_threshold=score_threshold,
                    top_n=len(documents),
                )
            
            return documents
```

```python
# ragflow/agent/tools/retrieval.py:94-265
class Retrieval(ToolBase):
    async def _retrieve_kb(self, query_text: str):
        # 1. 获取知识库
        kb_ids = self._get_dataset_ids()
        kbs = KnowledgebaseService.get_by_ids(filtered_kb_ids)
        
        # 2. 获取 embedding 和 rerank 模型
        embd_mdl = LLMBundle(tenant_id, embd_model_config)
        if self._param.rerank_id:
            rerank_mdl = LLMBundle(kbs[0].tenant_id, rerank_model_config)
        
        # 3. 主检索
        kbinfos = await settings.retriever.retrieval(
            query, embd_mdl, [kb.tenant_id for kb in kbs],
            filtered_kb_ids, 1, self._param.top_n,
            self._param.similarity_threshold,
            1 - self._param.keywords_similarity_weight,
            doc_ids=doc_ids, aggs=True, rerank_mdl=rerank_mdl,
            rank_feature=label_question(query, kbs),
        )
        
        # 4. TOC 增强
        if self._param.toc_enhance:
            cks = await settings.retriever.retrieval_by_toc(
                query, kbinfos["chunks"], [kb.tenant_id for kb in kbs],
                chat_mdl, self._param.top_n
            )
        
        # 5. 子块检索
        kbinfos["chunks"] = settings.retriever.retrieval_by_children(
            kbinfos["chunks"], [kb.tenant_id for kb in kbs]
        )
        
        # 6. KG 检索
        if self._param.use_kg:
            ck = await settings.kg_retriever.retrieval(
                query, [kb.tenant_id for kb in kbs], kb_ids,
                embd_mdl, chat_mdl
            )
```

```python
# qanything_kernel/core/retriever/parent_retriever.py:22-67
class SelfParentRetriever(ParentDocumentRetriever):
    async def _aget_relevant_documents(self, query, run_manager):
        # 1. 向量检索
        if self.search_type == "mmr":
            sub_docs = await self.vectorstore.amax_marginal_relevance_search(
                query, **self.search_kwargs
            )
        else:
            res = await self.vectorstore.asimilarity_search_with_score(
                query, **self.search_kwargs
            )
            scores = [score for _, score in res]
            sub_docs = [doc for doc, _ in res]
        
        # 2. 获取父文档 ID
        ids = [d.metadata[self.id_key] for d in sub_docs if self.id_key in d.metadata]
        
        # 3. 从文档存储获取父文档
        docs = await self.docstore.amget(ids)
        
        # 4. 合并分数
        for i, doc in enumerate(docs):
            if doc is not None:
                doc.metadata['score'] = scores[i]
        
        return [d for d in docs if d is not None]

# qanything_kernel/core/retriever/parent_retriever.py:213-247
class ParentRetriever:
    async def get_retrieved_documents(self, query, partition_keys, time_record, hybrid_search, top_k):
        # 1. Milvus 向量检索
        expr = f'kb_id in {partition_keys}'
        self.retriever.set_search_kwargs("similarity", k=top_k, expr=expr)
        query_docs = await self.retriever.aget_relevant_documents(query)
        
        # 2. ES 混合检索 (可选)
        if hybrid_search:
            filter = [{"terms": {"metadata.kb_id.keyword": partition_keys}}]
            es_sub_docs = await self.es_store.asimilarity_search(query, k=top_k, filter=filter)
            
            # 3. 合并结果
            milvus_doc_ids = [d.metadata[self.retriever.id_key] for d in query_docs]
            es_ids = [d.metadata[self.retriever.id_key] for d in es_sub_docs 
                     if d.metadata[self.retriever.id_key] not in milvus_doc_ids]
            es_docs = await self.retriever.docstore.amget(es_ids)
            query_docs.extend(es_docs)
        
        return query_docs
```

## 3. 组件架构差异分析

### 3.1 Dify - 工厂模式 + 策略模式

**优点**:
- 高度抽象，接口统一
- 易于扩展新的向量数据库 (只需实现 `AbstractVectorFactory`)
- 索引处理器工厂模式清晰
- 支持 10+ 种向量后端

**缺点**:
- 层次较多，代码跳转复杂
- 部分抽象层过厚

**扩展向量后端示例**:
```python
# dify/api/core/rag/datasource/vdb/my_vector.py
class MyVectorIndex(AbstractVectorFactory):
    def init_vector(self, dataset, attributes, embeddings):
        return MyVectorStore(dataset, attributes, embeddings)

# 注册到后端注册表
# vector_backend_registry.py
REGISTRY = {
    "my_vector": MyVectorIndex,
    ...
}
```

### 3.2 RAGFlow - 服务聚合模式

**优点**:
- 检索逻辑集中在 `Settings.retriever`
- 组件间解耦清晰
- 支持高级特性 (TOC/KG/跨语言)
- 支持 Infinity 和 Milvus 切换

**缺点**:
- 全局依赖较多
- 测试难度较大

### 3.3 QAnything - LangChain 继承模式

**优点**:
- 基于 LangChain `ParentDocumentRetriever`
- 代码简洁清晰
- 父子文档检索实现优雅

**缺点**:
- 依赖 LangChain 框架
- 灵活性受限

### 3.4 Coze Studio - DDD 分层模式

**优点**:
- DDD 分层清晰
- 领域边界明确
- 易于维护和测试

**缺点**:
- 层次较多
- 开发成本较高
