# coze-studio/backend/infra/document/parser/impl/builtin/parse_pdf.py
def extract_pdf_content(pdf_data: bytes, extract_images=True, extract_tables=True, filter_pages=None):
    """PDF 解析核心函数"""
    
    with pdfplumber.open(io.BytesIO(pdf_data)) as pdf:
        content = []
        
        for page_num, page in enumerate(pdf.pages):
            # 1. 提取文本
            text = page.extract_text(x_tolerance=2)
            content.append({
                'type': 'text',
                'content': text,
                'page': page_num + 1,
                'bbox': page.bbox
            })
            
            # 2. 提取图片 (可选)
            if extract_images:
                for img_index, img in enumerate(page.images):
                    try:
                        # 处理不同编码格式的图片
                        filters = img['stream'].get_filters()
                        data = img['stream'].get_data()
                        
                        if filters[-1][0] in LITERALS_DCT_DECODE:
                            # JPEG 处理
                            if LITERAL_DEVICE_CMYK in img['colorspace']:
                                i = Image.open(io.BytesIO(data))
                                i = ImageChops.invert(i)
                                i = i.convert("RGB")
                                i.save(buffered, format="PNG")
                            else:
                                buffered.write(data)
                        
                        # FLATE 解码处理
                        elif len(filters) == 1 and filters[0][0] in LITERALS_FLATE_DECODE:
                            width, height = img['srcsize']
                            channels = len(img['stream'].get_data()) / width / height / (img['bits'] / 8)
                            mode = ...
                            i = Image.frombytes(mode, img['srcsize'], data, "raw")
                            i.save(buffered, format="PNG")
                        
                        content.append({
                            'type': 'image',
                            'content': base64.b64encode(buffered.getvalue()).decode('utf-8'),
                            'page': page_num + 1,
                            'bbox': (img['x0'], img['top'], img['x1'], img['bottom'])
                        })
                    except Exception as err:
                        print(f"Skipping unsupported image: {err}")
            
            # 3. 提取表格 (可选)
            if extract_tables:
                tables = page.find_tables()
                for table in tables:
                    if is_structured_table(table):
                        # 转换为结构化数据
                        table_data = []
                        for row in table:
                            table_data.append([cell if cell else "" for cell in row])
                        content.append({
                            'type': 'table',
                            'data': table_data,
                            'page': page_num + 1
                        })
    
    return content

def is_structured_table(table):
    """判断是否为结构化表格"""
    if not table:
        return False
    row_count = len(table)
    col_count = max(len(row) for row in table)
    return row_count >= 2 and col_count >= 2
```

class PDFParser(BaseParser):
    def parse(self, content: bytes) -> Document:
        """
        轻量级 PDF 解析
        基于 pdfplumber
        专注于文本提取
        """
        
        # 使用 pdfplumber 提取文本
        with pdfplumber.open(io.BytesIO(content)) as pdf:
            text = ""
            for page in pdf.pages:
                text += page.extract_text() + "\n"
        
        return Document(content=text)
```

**特点**:
- 轻量级设计
- 基于 pdfplumber
- 专注于文本提取
- 无内置 OCR
